172 static const _BucketSearchResult searchBucketPos(apr_bucket_brigade* brigade, size_t startStrPos, size_t posFromStart) {
173  _BucketSearchResult result = {NULL, 0};
174  apr_bucket* bucket = APR_BRIGADE_FIRST(brigade);
175  const char* bucketStr = NULL;
176  apr_size_t bucketStrLen = 0;
177  apr_status_t status = apr_bucket_read(bucket, &bucketStr, &bucketStrLen, APR_BLOCK_READ);
178  //最初のbucketの文字列長を指定の長さが超えている場合はエラー
179  if (bucketStrLen < startStrPos) return result;
180  //最初のbucketで指定の長さを満たす場合
181  size_t len = bucketStrLen - startStrPos;
182  if (posFromStart < len) {
183  result.bucket = bucket;
184  result.bucketStrPos = startStrPos + posFromStart;
185  return result;
186  }
187  //目的の位置を探す
188  for (bucket = APR_BUCKET_NEXT(bucket); bucket != APR_BRIGADE_SENTINEL(brigade); bucket = APR_BUCKET_NEXT(bucket)) {
189  apr_status_t status = apr_bucket_read(bucket, &bucketStr, &bucketStrLen, APR_BLOCK_READ);
190  len += (size_t)bucketStrLen;
191  //このbucketに目的の位置がある場合
192  if (posFromStart < len) {
193  result.bucket = bucket;
194  result.bucketStrPos = bucketStrLen - (len - posFromStart);
195  return result;
196  }
197  }
198  //brigadeの最後まで達した。
199  return result;
200 }

256 const int CBucketController_pos(const CBucketController* p_this, const size_t pos) {
257  const CBucketController_Super* self = (const CBucketController_Super*)p_this;
258  //
259  //brigadeの終わりに来ている場合。
260  if (CBucketController_isEndBuffer((const CBucketController*)self)) return CBUCKET_CONTROLLER_ERR_POS;
261  //開始していない場合
262  if (self->currentBucketStrPos == -1) return CBUCKET_CONTROLLER_ERR_POS;
263  apr_bucket* bucket = self->currentBucket;
264  size_t len = self->currentBucketStrLen - self->currentBucketStrPos;
265  if (pos < len) {
266  return self->currentBucketStr[self->currentBucketStrPos + pos];
267  }
268  //
269  const char* bucketStr = NULL;
270  apr_size_t bucketStrLen = 0;
271  for (bucket = APR_BUCKET_NEXT(bucket); bucket != APR_BRIGADE_SENTINEL(self->inputBrigade); bucket = APR_BUCKET_NEXT(bucket)) {
272  apr_status_t status = apr_bucket_read(bucket, &bucketStr, &bucketStrLen, APR_BLOCK_READ);
273  //このbucketに目的の位置がある場合
274  if (pos < len + bucketStrLen) {
275  return bucketStr[pos - len];
276  }
277  len += (size_t)bucketStrLen;
278  }
279  //目的の位置が見つかる前にbucketがなくなった
280  return CBUCKET_CONTROLLER_ERR_POS;
281 }

284 AcBool CBucketController_forward(CBucketController* p_this) {
285  CBucketController_Super* self = (CBucketController_Super*)p_this;
286  //brigadeの終わりに来ている場合。

288  //

290  //カレントのbucketの中で移動ができた場合
291  if (self->currentBucketStrPos < self->currentBucketStrLen) {
292  return AC_TRUE;

294 
295  //次のbucketに移動
296  apr_bucket* bucket = self->currentBucket;
297  self->currentBucket = APR_BUCKET_NEXT(bucket);
298  BucketController_moveInputBucketToModifiedBrigade(self, bucket);
299  ++self->currentBucketStrPos;
300  //
301  if (APR_BRIGADE_EMPTY(self->inputBrigade) || APR_BUCKET_IS_EOS(self->currentBucket)) {
302  self->isEndBuffer = AC_TRUE;
303  return AC_FALSE;
304  }
305  return AC_TRUE;

309 AcBool CBucketController_execReplace(CBucketController* p_this, size_t len, const char* replaceStr) {
310  CBucketController_Super* self = (CBucketController_Super*)p_this;
311  //一度もforwardが呼ばれてない場合
312  if (self->currentBucketStrPos == -1) return AC_FALSE;
313  //inputBrigadeが空などバッファの終わりの場合
314  if (CBucketController_isEndBuffer((CBucketController*)self)) return AC_FALSE;
315  apr_bucket* b = self->currentBucket;
316  //置換開始位置のsplit(カレントの先頭の場合は分割不要のため何もしない)
317  if (self->currentBucketStrPos != 0) {
318  apr_bucket_split(b, self->currentBucketStrPos);
319  //置換対象じゃない文字列を最後尾に追加
320  BucketController_moveInputBucketToModifiedBrigade(self, APR_BRIGADE_FIRST(self->inputBrigade));
321  }
322 
323  //長さ0の置換の場合
324  if (len == 0) {
325  //置換文字列を最後尾に追加。
326  apr_bucket* tmpb = apr_bucket_transient_create(replaceStr, strlen(replaceStr), self->bucketAlloc);
327  APR_BRIGADE_INSERT_TAIL(self->modifiedBrigate, tmpb);
328  //bucket位置や文字列の位置等の初期化
329  BucketController_initCurrentVar(self, APR_BRIGADE_FIRST(self->inputBrigade));
330  return AC_TRUE;
331  }
332  //置換終了位置を検索する
333  const _BucketSearchResult result = searchBucketPos(self->inputBrigade, 0, len - 1);
334  //位置が見つからなかった場合（バッファのエンドに達した場合など）
335  if (result.bucket == NULL) {
336  //bucket位置や文字列の位置等の初期化
337  BucketController_initCurrentVar(self, APR_BRIGADE_FIRST(self->inputBrigade));
338  return AC_FALSE;
339  }
340 
341  //置換文字列を最後尾に追加。
342  apr_bucket* tmpb = apr_bucket_transient_create(replaceStr, strlen(replaceStr), self->bucketAlloc);
343  APR_BRIGADE_INSERT_TAIL(self->modifiedBrigate, tmpb);
344 
345  //置換対象bucketを削除する
346  while ((b = APR_BRIGADE_FIRST(self->inputBrigade)) != result.bucket) {
347  //先頭のbucketを削除する
348  apr_bucket_delete(b);
349  }
350  //inputBrigadeの先頭のbucketを分割する
351  apr_bucket_split(b, result.bucketStrPos + 1);
352  //分割したbucketを削除する（置換対象）
353  apr_bucket_delete(b);
354  //bucket位置や文字列の位置等の初期化
355  BucketController_initCurrentVar(self, APR_BRIGADE_FIRST(self->inputBrigade));
356  //先頭のbucketが0文字の場合、削除する
357  if (self->currentBucketStrLen == 0 && self->currentBucketStr != NULL) {
358  apr_bucket_delete(self->currentBucket);
359  BucketController_initCurrentVar(self, APR_BRIGADE_FIRST(self->inputBrigade));
360  }
361 
362  return AC_TRUE;
363 }

819 apr_bucket_brigade* CAnalysisManager_run(CAnalysisManager* p_this, apr_bucket_brigade* inputBrigade) {
820  CAnalysisManager_Super* self = (CAnalysisManager_Super*)p_this;
821  //
822  if (self->isAllPassThough) {
823  return CAnalysisManager_PassThrough(self, inputBrigade);
824  }
825  if (inputBrigade != NULL) {
826  CBucketController_addBrigade(self->bucketController, inputBrigade);
827  //CANALYSIS_MANAGER_DEBUG_PRINT(p_this);
828  }
829  //
830  AcBool bl = AC_FALSE;
831  apr_bucket_brigade* brigade = NULL;
832  AnalysisCommandType preType = AC_FORWARD;
833  while (!CBucketController_isEndBuffer(self->bucketController)) {
834  brigade = NULL;
835  AnalysisCommandType cmdType = self->nextCmd.type;
836  switch (cmdType) {
837  case AC_FORWARD:
838  brigade = CAnalysisManager_forward(self, self->nextCmd);
839  break;
840  case AC_FORWARD_N:
841  brigade = CAnalysisManager_forward_n(self, self->nextCmd);
842  break;
843  case AC_POS:
844  bl = CAnalysisManager_pos(self, self->nextCmd);
845  if(!bl){
846  brigade = CBucketController_exportModifiedBrigade(self->bucketController);
847  }
848  break;
849  case AC_REPLACE:
850  if(preType == AC_REPLACE){
851  //連続でREPLACEが実行されようとした場合、強引にエラーを出させてexitさせる
852  ac_checkClass("err", "", "The AC_REPLACE was called multiple times in series at CAnalysisManager_run.", AC_TRUE);
853  }
854  CAnalysisManager_replace(self, self->nextCmd);
855  break;
856  case AC_PASS_THROUGH_ALL:
857  self->isAllPassThough = AC_TRUE;
858  brigade = CAnalysisManager_PassThrough(self, inputBrigade);
859  }
860  preType = cmdType;
861 
862  //終了の場合
863  if (CBucketController_isEOS(self->bucketController)) {
864  brigade = CAnalysisManager_end(self, self->nextCmd, brigade);
865  }
866  if (brigade != NULL) return brigade;
867  }
868  return CBucketController_exportModifiedBrigade(self->bucketController);
869 }
