100 static void CAnalysisParser_Split_acceptFunc(CAnalysisParser* p_this, const char c){
101  CAnalysisParser_Split_Super* self = (CAnalysisParser_Split_Super*)p_this;
102  //
103  if(self->parentMember.isEnd) return;
104 
105  //ストックのバッファサイズを超えない場合のみストック文字列をセットする
106  if(CAnalysisParser_Split_getStockStrLen(self) + 1 <= self->stockStrBufferSize){
107  //ストック文字列をセット
108  *self->stockStrPos = c;
109  ++self->stockStrPos;
110  }
111 
112  //チェックする
113  if(self->parentMember.status == AnalysisParserStatus_Searching){
114  //開始文字と違った場合、次の文字へ。
115  if(c != *self->delimiterStrPos) return;
116  //開始文字と一致した場合、ステータスを変える
117  self->parentMember.status = AnalysisParserStatus_Matching;
118  } else if(self->parentMember.status == AnalysisParserStatus_Matching){
119  //文字列とマッチしなかった場合、ステータスをリセットして次へ。
120  if(c != *self->delimiterStrPos){
121  CAnalysisParser_Split_resetFunc((CAnalysisParser*)self);
122  return;
123  }
124  }
125 
126  //マッチ対象の次の文字へ進めておく
127  ++self->delimiterStrPos;
128  //文字列全てマッチした場合、ステータスを終了状態に変更
129  if(*self->delimiterStrPos == '¥0'){
130  self->parentMember.isEnd = AC_TRUE;
131  self->parentMember.status = AnalysisParserStatus_Matched;
132  *self->stockStrPos = '¥0';
133  }
134 }

315 static void CAnalysisParser_HttpHeader_acceptFunc(CAnalysisParser* p_this, const char c){
316  CAnalysisParser_HttpHeader_Super* self = (CAnalysisParser_HttpHeader_Super*)p_this;
317  //
318  if(self->parentMember.isEnd) return;
319  //マッチ確認中
320  if(self->status == HttpHeaderStatus_Attribute){
321  //属性名を検索する
322  CAnalysisParser* parser = (CAnalysisParser*)self->attrNameParser;
323  CAnalysisParser_accept(parser, c);
324  if(CAnalysisParser_isEnd(parser)){
325  if(CAnalysisParser_isMatched(parser)){
326  //属性名が指定の名前だった場合、値をパースするステータスに変更する
327  self->status = HttpHeaderStatus_Value;
328  } else{
329  //属性名が指定のものではなかった場合、行の最後を探す
330  self->status = HttpHeaderStatus_HeaderEndSearching;
331  }
332  }
333  } else if(self->status == HttpHeaderStatus_Value){
334  //属性名がマッチしたので、値をコピーする
335  CAnalysisParser* parser = (CAnalysisParser*)self->valueParser;
336  CAnalysisParser_accept(parser, c);
337  if(CAnalysisParser_isEnd(parser)){
338  //パースを終わりにする
339  self->parentMember.status = AnalysisParserStatus_Matched;
340  self->parentMember.isEnd = AC_TRUE;
341  }
342  }
343 
344  //
345  if(self->status == HttpHeaderStatus_HeaderEndSearching){
346  //属性名がマッチしなかったので値は無視して、行の終わり(¥r)を探す。
347  if(c == '¥r'){
348  //¥nを探す（この方法だと¥rX¥nのような文字列の場合もヒットしてしまうが悪意のある場合のみなので良しとする）
349  self->status = HttpHeaderStatus_HeaderEndMatching;
350  }
351  } else if(self->status == HttpHeaderStatus_HeaderEndMatching){
352  //行の終わり(¥n)を探す。
353  if(c == '¥n'){
354  //パースを終わりにする
355  self->parentMember.isEnd = AC_TRUE;
356  }
357  }
358 }
