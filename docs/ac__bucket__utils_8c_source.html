370 char** ac_splitKeyValueArrayWithQuote(const char* str, const char* seps, AcBool isSplitEq, size_t maxSize){
371 
372  char** ret = (char**)malloc(sizeof(char*) * (maxSize * 2 + 1) + strlen(str) + 1);
373  if(ret == NULL) return NULL;
374  //
375  char* copiedStr = ((char*)ret + sizeof(char*) * (maxSize * 2 + 1));
376  strcpy(copiedStr, str);
377  char* emptyStr = copiedStr + strlen(copiedStr);
378  char** startPos = ret;
379  *startPos = copiedStr;
380  *(++startPos) = emptyStr;
381  int cnt = 0;
382  for(char* p = copiedStr; *p != '¥0'; ++p){
383  if(*p == '"'){
384  ++p;
385  //対になる"を探す
386  for(; *p != '¥0'; ++p){
387  if(*p == '¥¥'){
388  if(++p == '¥0') break;
389  } else if(*p == '"'){
390  break;
391  }
392  }
393  } else if(strchr(seps, *p) != NULL){
394  //区切り文字だった場合
395  *p = '¥0';
396  //blank以外の場合だけ登録する
397  if(*(p - 1) != '¥0'){
398  ++cnt;
399  if(cnt >= maxSize) break;
400  *(++startPos) = p + 1;
401  *(++startPos) = emptyStr;
402  }
403  }
404  }
405 
406  //配列の最後が空文字なら無視する
407  if(**(startPos - 1) == '¥0') startPos -= 2;
408 
409  //配列の最後にNULLを設定

411 
412  //"="を区切るかどうか？
413  if(!isSplitEq) return ret;
414

417  for(char* a = *startPos; *a != '¥0'; ++a){
