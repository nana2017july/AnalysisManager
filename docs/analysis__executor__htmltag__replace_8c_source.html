113 static const AnalysisCommand CAnalysisExecutor_HtmlTagReplace_forward(CAnalysisExecutor* p_this,
114  const AnalysisCommand cmd, const AcBool isRejectPreCmd, const char c)
115 {
116  CAnalysisExecutor_HtmlTagReplace_Super* self = (CAnalysisExecutor_HtmlTagReplace_Super*)p_this;
117 
118  //前回のコマンドを実行拒否されたので状態リセットする
119  if(isRejectPreCmd){
120  //リセット
121  reset(self);
122  }
123  //
124  AnalysisCommand nextCmd;
125 
126  //見つかったタグの後ろの位置まで進める場合。
127  if(self->replaceStatus == HtmlTagReplaceStatus_SearchTagEndWithReplace
128  || self->replaceStatus == HtmlTagReplaceStatus_SearchTagEndWithNoReplace)
129  {
130  //置換位置（タグの終わりの次の位置）まで来たので置換する
131  if(self->replaceStatus == HtmlTagReplaceStatus_SearchTagEndWithReplace){
132  //置換する場合
133  nextCmd.type = AC_REPLACE;
134  nextCmd.replace.replacement = self->replaceStr;
135  nextCmd.replace.len = 0;
136  //ステータスを変更（タグの開始を探すステータスに）
137  self->replaceStatus = HtmlTagReplaceStatus_SearchStart;
138  return nextCmd;
139  }
140 
141  //状態をリセットして下の処理に任せる
142  reset(self);
143  }
144 
145  //タグの開始位置を探す場合
146  //タグの開始ではなかった場合
147  if(c != '<'){
148  nextCmd.type = AC_FORWARD;
149  return nextCmd;
150  }
151 
152  //タグの開始文字だった場合、AC_POSでタグの終了位置を検索する
153  self->replaceStatus = HtmlTagReplaceStatus_MatchingTagName;
154  nextCmd.type = AC_POS;
155  nextCmd.pos.pos = 1;
156  //検査位置をリセット
157  self->tagNameStrPos = self->tagName;
158  self->tagStrLen = 1;
159  return nextCmd;
160 }

179 static const AnalysisCommand CAnalysisExecutor_HtmlTagReplace_pos(CAnalysisExecutor* p_this,
180  const AnalysisCommand cmd, const char c, const AcBool isEos)
181 {
182  CAnalysisExecutor_HtmlTagReplace_Super* self = (CAnalysisExecutor_HtmlTagReplace_Super*)p_this;
183  //
184  AnalysisCommand nextCmd;
185 
186  ++self->tagStrLen;
187 
188  //EoSだった場合でこの関数に来た場合、マッチもタグの終了も見つかってないので読み飛ばして終了させる。
189  if(isEos){
190  nextCmd.type = AC_FORWARD_N;
191  nextCmd.forward_n.len = cmd.pos.pos;
192  return nextCmd;
193  }
194 
195  //タグ名のマッチングを検査している場合
196  if(self->replaceStatus == HtmlTagReplaceStatus_MatchingTagName){
197  if(tolower(c) == *self->tagNameStrPos){
198  //1文字マッチした場合
199  ++self->tagNameStrPos;
200  //タグが完全マッチした場合
201  if(*self->tagNameStrPos == '¥0'){
202  self->replaceStatus = HtmlTagReplaceStatus_MatchingAfterTagName;
203  }
204  } else{
205  //マッチしなかった場合、タグの終了を探す
206  self->replaceStatus = HtmlTagReplaceStatus_SearchTagEndWithNoReplace;
207  }
208  //次の位置の文字を読み取る
209  nextCmd.type = AC_POS;
210  nextCmd.pos.pos = cmd.pos.pos + 1;
211  return nextCmd;
212  }
213 
214  //タグ名の後ろの文字がスペースや＞かを検査している場合
215  if(self->replaceStatus == HtmlTagReplaceStatus_MatchingAfterTagName){
216  //タグの終了の場合
217  if(c == '>'){
218  //ステータスを変えて、後ろの処理に任せる
219  self->replaceStatus = HtmlTagReplaceStatus_SearchTagEndWithReplace;
220  } else{
221  //スペースの場合
222  if(isspace(c)){
223  //指定タグにマッチ＆タグの終了位置を探すステータスに変えて、後ろの処理に任せる
224  self->replaceStatus = HtmlTagReplaceStatus_SearchTagEndWithReplace;
225  } else{
226  //指定タグにマッチしなかった＆タグの終了位置を探すステータスに変えて、後ろの処理に任せる
227  self->replaceStatus = HtmlTagReplaceStatus_SearchTagEndWithNoReplace;
228  }
229  }
230  }
231 
232  //タグがマッチせず、タグの終了位置を探している場合
233  if(self->replaceStatus == HtmlTagReplaceStatus_SearchTagEndWithNoReplace){
234  if(c == '>'){
235  nextCmd.type = AC_FORWARD_N;
236  nextCmd.forward_n.len = self->tagStrLen;
237  self->tagStrLen = 0;
238  } else{
239  nextCmd.type = AC_POS;
240  nextCmd.pos.pos = cmd.pos.pos + 1;
241  }
242  return nextCmd;
243  }
244 
245  //タグがマッチして、タグの終了位置を探している場合
246  if(self->replaceStatus == HtmlTagReplaceStatus_SearchTagEndWithReplace){
247  if(c != '>'){//タグの終了ではない場合
248  nextCmd.type = AC_POS;
249  nextCmd.pos.pos = cmd.pos.pos + 1;
250  }else{//タグの終了の場合
251  if(self->isTagReplace){
252  //タグ自体を置換する場合
253  nextCmd.type = AC_REPLACE;
254  nextCmd.replace.replacement = self->replaceStr;
255  nextCmd.replace.len = self->tagStrLen;
256  } else{
257  //タグの後ろに文字列を挿入する場合
258  nextCmd.type = AC_FORWARD_N;
259  nextCmd.forward_n.len = self->tagStrLen;
260  }
261  }
262  return nextCmd;
263  }
264 
265  //ここには来るはずないのでエラー処理をする
266  nextCmd.type = AC_PASS_THROUGH_ALL;
267  fprintf(stderr, "Error in CAnalysisExecutor_HtmlTagReplace_pos");
268  return nextCmd;
269 }
