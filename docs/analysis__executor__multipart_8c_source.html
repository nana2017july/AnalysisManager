159 static const AnalysisCommand CAnalysisExecutor_Multipart_pos(CAnalysisExecutor* p_this,
160  const AnalysisCommand cmd, const char c, const AcBool isEos)
161 {
162  CAnalysisExecutor_Multipart_Super* self = (CAnalysisExecutor_Multipart_Super*)p_this;
163  //
164  AnalysisCommand nextCmd;
165  const char* str;
166  size_t len;
167 
168  //EOSの場合、読み飛ばして終了させる（不完全な形なので）
169  if(isEos){
170  nextCmd.type = AC_FORWARD_N;
171  nextCmd.forward_n.len = cmd.pos.pos;
172  return nextCmd;
173  }
174 
175  //パーサに文字を渡す
176  if(self->status == MultipartStatus_Header){
177  //ヘッダ部の解析をしている場合
178  CAnalysisParser* parser = (CAnalysisParser*)self->headerParser;
179  CAnalysisParser_accept(parser, c);
180  if(CAnalysisParser_isEnd(parser)){
181  if(CAnalysisParser_isMatched(parser)){
182  //属性がマッチしていたら、名前を取得する
183  CAnalysisParser_HttpHeader_getValue((CAnalysisParser_HttpHeader*)parser, &str, &len);
184  char** ret = ac_splitHttpHeaderValue(str, ";", AC_TRUE, 4);
185  AcBool isValueFile = AC_FALSE;
186  for(char** p = ret; *p != NULL; p += 2){
187  //"name"にマッチしたらコピー
188  if(strcmp("name", *p) == 0){
189  size_t nameLen = strlen(*(p + 1));
190  if(nameLen >= MULTIPART_NAME_MAX_LEN) nameLen = MULTIPART_NAME_MAX_LEN;
191  strncpy(self->name, *(p + 1), nameLen);
192  self->name[nameLen] = '¥0';
193  //ac_toLowerCase(self->name);
194  } else if(strcmp("filename", *p) == 0){
195  //ファイルの場合
196  isValueFile = AC_TRUE;
197  }
198  }
199  //nameがマッチして、valueがファイルの場合に設定
200  if(*self->name != '¥0' && isValueFile) self->isValueFile = AC_TRUE;
201  //解放
202  free((void*)ret);
203  //ヘッダパーサをリセットして次のヘッダへ
204  CAnalysisParser_reset(parser);
205 
206  } else if(CAnalysisParser_getParsedStrLen(parser) == 2){
207  //ヘッダの終わり（¥r¥n）だった場合
208  self->status = MultipartStatus_Body;
209  } else{
210  //行の終わりまで来ているがマッチしなかった場合、ヘッダパーサをリセットして次のヘッダへ
211  CAnalysisParser_reset(parser);
212  }
213  }
214  } else if(self->status == MultipartStatus_Body){
215  //ボディ部のパースをしている場合
216  CAnalysisParser* parser = (CAnalysisParser*)self->boundaryParser;
217  CAnalysisParser_accept(parser, c);
218  if(CAnalysisParser_isEnd(parser)){
219  //マッチしたかEOSの場合
220  if(!self->isParamsAddedIfNotFile || !self->isValueFile){
221  //パラメタテーブルに追加する
222  CAnalysisParser_Split_getStockStr((CAnalysisParser_Split*)parser, &str, &len);
223  //value文字列をコピーする
224  size_t valueLen = len;
225  if((int)len - (self->boundaryStrLen) >= 0) valueLen = len - (self->boundaryStrLen);
226  if(valueLen >= MULTIPART_VALUE_LEN) valueLen = MULTIPART_VALUE_LEN;
227  strncpy(self->value, str, valueLen);
228  self->value[valueLen] = '¥0';
229  //テーブルにパラメタ名と値を保存(tableのkeyは大文字小文字を区別しない)
230  apr_table_add(self->paramsTable, self->name, self->value);
231  //クラスをリセット
232  CAnalysisExecutor_Multipart_reset(self);
233  }
234  //次のブロックまで読み飛ばす(パースした部分を読み飛ばす)
235  nextCmd.type = AC_FORWARD_N;
236  nextCmd.forward_n.len = cmd.pos.pos + 2 + 1; //改行も読み飛ばす
237  return nextCmd;
238  }
239 
240  }
241 
242  nextCmd.type = AC_POS;
243  nextCmd.pos.pos = cmd.pos.pos + 1;
244  return nextCmd;
245 }
